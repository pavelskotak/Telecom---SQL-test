--- The following query is from SAP HANA but applies to any DB
--- Do not worry if the tables/columns are not familiar to you 
----   -> you do not need to interpret the result (in fact the query does not reflect actual DB content)
SELECT 
	RSEG.EBELN,
	RSEG.EBELP,
    RSEG.BELNR,
    RSEG.AUGBL AS AUGBL_W,
    LPAD(EKPO.BSART,6,0) as BSART,
	BKPF.GJAHR,
	BSEG.BUKRS,
	BSEG.BUZEI,
	BSEG.BSCHL,
	BSEG.SHKZG,
    CASE WHEN BSEG.SHKZG = 'H' THEN (-1) * BSEG.DMBTR ELSE BSEG.DMBTR END AS DMBTR, --  lepsi pouzit decode(BSEG.SHKZG, 'H', BSEG.SHKZG * (-1), BSEG.SHKZG) END AS DMBTR, 
    COALESCE(BSEG.AUFNR, 'Kein SM-A Zuordnung') AS AUFNR, -- radeji pouzit nvl() funkci, bseg.aufnr nemusi byt varchar 
    COALESCE(LFA1.LAND1, 'Andere') AS LAND1, -- radeji pouzit nvl() funkci, LFA1.LAND1 nemusi byt varchar 
    LFA1.LIFNR,
    LFA1.ZSYSNAME,
    BKPF.BLART as BLART,
    BKPF.BUDAT as BUDAT,
    BKPF.CPUDT as CPUDT
FROM "DTAG_DEV_CSBI_CELONIS_DATA"."dtag.dev.csbi.celonis.data.elog::V_RSEG" AS RSEG
LEFT JOIN "DTAG_DEV_CSBI_CELONIS_WORK"."dtag.dev.csbi.celonis.app.p2p_elog::__P2P_REF_CASES" AS EKPO ON 1=1 -- neni potřeba používat klič 1 = 1
    AND RSEG.ZSYSNAME = EKPO.SOURCE_SYSTEM
    AND RSEG.MANDT = EKPO.MANDT
    AND RSEG.EBELN || RSEG.EBELP = EKPO.EBELN || EKPO.EBELP -- využil bych concat() funkci, cely left join je lepsi dat az za všechny inner joiny
INNER JOIN "DTAG_DEV_CSBI_CELONIS_DATA"."dtag.dev.csbi.celonis.data.elog::V_BKPF" AS BKPF ON 1=1 -- neni potřeba používat klíč 1 = 1
    AND BKPF.AWKEY = RSEG.AWKEY
    AND RSEG.ZSYSNAME = BKPF.ZSYSNAME
    AND RSEG.MANDT in ('200') -- lepší použít "=" místo "in" 
INNER JOIN "DTAG_DEV_CSBI_CELONIS_DATA"."dtag.dev.csbi.celonis.data.elog::V_BSEG" AS BSEG ON 1=1 -- není potřeba používat klíč 1 = 1 
    AND DATS_IS_VALID(BSEG.ZFBDT) = 1 -- pokud neni funkce vytvořená (možná v SAP HANA je - nevím), tak script spadne 
    AND BSEG.KOART = 'K'
    AND CAST(BSEG.GJAHR AS INT) = 2020 -- lepsi pouzit to_number() funkci
    AND BKPF.ZSYSNAME = BSEG.ZSYSNAME
    AND BKPF.MANDT = BSEG.MANDT
    AND BKPF.BUKRS = BSEG.BUKRS
    AND BKPF.GJAHR = BSEG.GJAHR
    AND BKPF.BELNR = BSEG.BELNR
    AND BSEG.DMBTR*-1 >= 0 -- -1 by mělo být v závorce (v některých databázích spadne, někde ne)
INNER JOIN (SELECT * FROM "DTAG_DEV_CSBI_CELONIS_DATA"."dtag.dev.csbi.celonis.data.elog::V_LFA1" AS TEMP 
            WHERE TEMP.LIFNR > '020000000') AS LFA1 ON 1=1 -- není nutné využívát vnořený select, stačí inner join a přidat podmínku u joinu "and TEMP.LIFNR > '020000000'"
    AND BSEG.ZSYSNAME = LFA1.ZSYSNAME
    AND BSEG.LIFNR=LFA1.LIFNR
    AND BSEG.MANDT=LFA1.MANDT
    AND LFA1.LAND1 in ('DE','SK')
;
